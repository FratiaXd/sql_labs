  2  customerName VARCHAR(20),

  4  customerCity VARCHAR(20),
  5  PRIMARY KEY (customerName));

CREATE TABLE branch(
  2  branchName VARCHAR(20),
  3  branchCity VARCHAR(20),
  4  PRIMARY KEY (branchName));

CREATE TABLE deposit(
  2  customerName VARCHAR(20),
  3  branchName VARCHAR(20),
  4  accountNumber INT,
  5  balance DECIMAL(7,2) DEFAULT '0',
  6  PRIMARY KEY (accountNumber),
  7  CONSTRAINT customerBr FOREIGN KEY (branchName)
  8  REFERENCES branch(branchName),
  9  CONSTRAINT customerNm FOREIGN KEY (customerName)
 10  REFERENCES customer(customerName));

CREATE TABLE loan(
  2  customerName VARCHAR(20),
  3  branchName VARCHAR(20),
  4  loanNumber INT,
  5  amount DECIMAL(7,2) DEFAULT '0',
  6  PRIMARY KEY (loanNumber),
  7  CONSTRAINT branchLn FOREIGN KEY (branchName)
  8  REFERENCES branch(branchName),
  9  CONSTRAINT customerLn FOREIGN KEY (customerName)
 10  REFERENCES customer(customerName));
